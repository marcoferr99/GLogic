\section{Simply Typed Lambda Calculus}


\subsection{Syntax}

Taking inspiration from \cite{church:types} and \cite{pierce:plf} we define the types of the simply typed lambda calculus as:
\[
    \inddef{T}{\typrp \mid \tyind \mid \tyarr{T}{T}}
\]
That is, $\typrp$ is a type, $\tyind$ is a type and given two types $A$ and $B$, the expression $A \rightarrow B$ is also a type.
The type $\typrp$ represents the type of propositions, $\tyind$ represents the type of individuals and $A \rightarrow B$ represents the type of functions from $A$ to $B$.

Some examples of types:
\[
    \typrp \qquad
    \tyarr{\typrp}{\tyind} \qquad
    \tyarr{\typrp}{\tyarr{\tyind}{\typrp}} \qquad
    \tyarr{(\tyarr{\typrp}{\tyind})}{\typrp}
\]
We consider $\tyarr{}{}$ to be right associative, and thus $\tyarr{\typrp}{\tyarr{\tyind}{\typrp}}$ is the same as $\tyarr{\typrp}{(\tyarr{\tyind}{\typrp})}$.

We define the terms of the simply typed lambda calculus (using the Bruijn levels) as:
\[
    \inddef{t}{\lvln \mid \tmapp{t}{t} \mid \tmabs{T}{t} \mid \tmnom{T}{n} \mid \alpha \mid \tmfors{T} \mid \tmexs{T} \mid \tmnabs{T}}
\]
where $\lvln$ is a natural number, $T$ is a type and $\alpha$ is one of the following symbols:
\[
    \inddef{\alpha}{\bot \mid \top \mid \land \mid \lor \mid \tmimp{}{}}
\]
Some examples of terms:
\[
    \tmabs{\tyind}{0} \qquad
    \tmapp{(\tmabs{\tyarr{\tyind}{\typrp}}{2})}{1} \qquad
    \tmabs{\typrp}{\tmapp{3}{\lor}} \qquad
    \tmabs{\tyarr{\typrp}{\typrp}}{\tmabs{\tyind}{\tmapp{0}{\top}}} \qquad
    \tmapp{\tmfors{o}}{\bot}
\]

The interpretation of a term depends on a fixed number $m$ of free variables.
All the levels which are less then $m$ represent free variables, while the others represent bound variables.
For example, if the free variables are $2$ (that is, the levels $0$ and $1$), the terms $\tmabs{\tyind}{0}$ and $\tmapp{(\tmabs{\tyarr{\tyind}{\typrp}}{2})}{1}$ can be interpreted respectively as $\lambda \, x \colon \tyind, y$ and $\tmapp{(\lambda \, x \colon \tyarr{\tyind}{\typrp}, x)}{y}$.
If the free variables are $0$, then $\tmabs{\tyind}{0}$ corresponds to $\lambda \, x \colon \tyind, x$, while the term $\tmapp{(\tmabs{\tyarr{\tyind}{\typrp}}{2})}{1}$ makes no sense, because $1$ and $2$ should be two distinct bound variables.


\subsection{Typing}

Given a list of types $\Gamma$, a natural number $n$ and a type $T$, we denote the $n$-th element of $\Gamma$ by $\nth{\Gamma}{n}$ and the list constructed by appending the element $T$ to the end of $\Gamma$ as $\listappel{\Gamma}{T}$.

\begin{definition}
    Given a list of types $\Gamma$, a term $t$ and a type $T$, we define the proposition $\hastype{\Gamma}{t}{T}$ by the following inference rules:
    \begin{gather*}
        \inferrule{ }{\hastype{\Gamma}{\lvln}{\nth{\Gamma}{\lvln}}} \\
        %
        \inferrule{\hastype{\Gamma}{s}{\tyarr{A}{B}} \\ \hastype{\Gamma}{t}{A}}{\hastype{\Gamma}{\tmapp{s}{t}}{B}} \\
        %
        \inferrule{\hastype{\listappel{\Gamma}{A}}{t}{B}}{\hastype{\Gamma}{\tmabs{A}{t}}{\tyarr{A}{B}}} \\
        %
        \inferrule{ }{\hastype{\Gamma}{\bot}{\typrp}} \qquad
        \inferrule{ }{\hastype{\Gamma}{\top}{\typrp}} \\
        %
        \inferrule{ }{\hastype{\Gamma}{\land}{\tyarr{\typrp}{\tyarr{\typrp}{\typrp}}}} \qquad
        \inferrule{ }{\hastype{\Gamma}{\lor}{\tyarr{\typrp}{\tyarr{\typrp}{\typrp}}}} \\
        %
        \inferrule{ }{\hastype{\Gamma}{\tmimp{\,}{\,}}{\tyarr{\typrp}{\tyarr{\typrp}{\typrp}}}} \\
        %
        \inferrule{ }{\hastype{\Gamma}{\tmfors{T}}{\tyarr{(\tyarr{T}{\typrp})}{\typrp}}} \\
        %
        \inferrule{ }{\hastype{\Gamma}{\tmexs{T}}{\tyarr{(\tyarr{T}{\typrp})}{\typrp}}} \\
        %
        \inferrule{ }{\hastype{\Gamma}{\tmnabs{T}}{\tyarr{(\tyarr{T}{\typrp})}{\typrp}}}
    \end{gather*}
    If the proposition $\hastype{\Gamma}{t}{T}$ holds, we say that $t$ \emph{has type} $T$ in the \emph{context} $\Gamma$.
\end{definition}

Some examples:
\begin{gather*}
    \inferrule{ }{\hastype{\listdelim{\tyind \listsep \typrp}}{0}{\tyind}} \\
    %
    \inferrule{
        \inferrule{
            \inferrule{
                \inferrule{ }{\hastype{\listdelim{\tyarr{\tyind}{\typrp} \listsep \tyind}}{0}{\tyarr{\tyind}{\typrp}}} \\
                \inferrule{ }{\hastype{\listdelim{\tyarr{\tyind}{\typrp} \listsep \tyind}}{1}{\tyind}}
            }{
                \hastype{\listdelim{\tyarr{\tyind}{\typrp} \listsep \tyind}}{\tmapp{0}{1}}{\typrp}
            }
        }{
            \hastype{\tyarr{\tyind}{\typrp}}{\tmabs{\tyind}{\tmapp{0}{1}}}{\tyarr{\tyind}{\typrp}}
        }
    }{
        \hastype{}{\tmabs{\tyarr{\tyind}{\typrp}}{\tmabs{\tyind}{\tmapp{0}{1}}}}{\tyarr{(\tyarr{\tyind}{\typrp})}{\tyarr{\tyind}{\typrp}}}
    }
\end{gather*}

We could, for example, define $s \lor t$ as a ``primitive'' term, with the typing rule:
\[
    \inferrule{\hastype{\Gamma}{s}{\typrp} \\ \hastype{\Gamma}{t}{\typrp}}{\hastype{\Gamma}{s \lor t}{\typrp}} \\
\]
and similarly for all the other constant terms.
However, especially in formalization, it is easier to have a small core of basic terms to prove properties of (levels, abstraction and application), and have any additional terms simply as constants.


\subsection{Substitution}

The substitution of the terms $s_0, \dots, s_n$ with context $m$ into the term $t$, indicated as $\subst{t}{m}{s_0, \dots, s_n}$, is defined recursively as follows:
\begin{gather*}
    \inddef{\subst{i}{m}{s_0, \dots, s_n}}{s_i} \infcondr{if $i \in \mathbb{N}$ and $i \le n$} \\
    \inddef{\subst{i}{m}{s_0, \dots, s_n}}{i} \infcondr{if $i \in \mathbb{N}$ and $i > n$} \\
    %
    \inddef{\subst{(\tmapp{t_1}{t_2})}{m}{s_0, \dots, s_n}}{\tmapp{(\subst{t_1}{m}{s_0, \dots, s_n})}{(\subst{t_2}{m}{s_0, \dots, s_n})}} \\
    %
    \inddef{\subst{(\tmabs{T}{t})}{m}{s_0, \dots, s_n}}{\tmabs{T}{(\subst{t}{m+1}{\incrbound{s_0}, \dots, \incrbound{s_n}, m})}} \\
    %
    \inddef{\subst{t}{m}{s_0, \dots, s_n}}{t} \infcondr{in all other cases}
\end{gather*}
where $\incrbound{s_k}$ is the term obtained from $s_k$ by increasing all of its bound levels by one, with respect to a context with $m$ free variables.

If we had defined terms like $s \lor t$ as primitive, we would have had to give explicit rules for them in the substitution definition.

Some examples:
\begin{gather*}
    \subst{(\tmapp{1}{0})}{2}{\tmabs{\tyind}{0} \listsep 1} =
    \tmapp{(\subst{1}{2}{\tmabs{\tyind}{0} \listsep 1})}{(\subst{0}{2}{\tmabs{\tyind}{0} \listsep 1})} =
    \tmapp{1}{(\tmabs{\tyind}{0})} \\
    %
    \subst{(\tmabs{\typrp}{\tmapp{1}{0}})}{2}{\tmabs{\tyind}{0} \listsep 1} =
    \tmabs{\typrp}{(\subst{(\tmapp{1}{0})}{3}{\tmabs{\tyind}{0} \listsep 1 \listsep 2})} =
    \tmabs{\typrp}{\tmapp{1}{(\tmabs{\tyind}{0})}} \\
    %
    \subst{(\tmabs{\typrp}{\tmapp{1}{0}})}{2}{\tmabs{\tyind}{0}} =
    \tmabs{\typrp}{(\subst{(\tmapp{1}{0})}{3}{\tmabs{\tyind}{0} \listsep 2})} =
    \tmabs{\typrp}{\tmapp{2}{(\tmabs{\tyind}{0})}} \\
    %
    \subst{(\tmabs{\typrp}{0})}{2}{\tmabs{\tyind}{\tmapp{0}{1}}} =
    \tmabs{\typrp}{(\subst{0}{3}{\tmabs{\tyind}{\tmapp{0}{1}} \listsep 2})} =
    \tmabs{\typrp}{\tmabs{\tyind}{\tmapp{0}{1}}} \\
    %
    \subst{(\tmabs{\typrp}{0})}{1}{\tmabs{\tyind}{\tmapp{0}{1}}} =
    \tmabs{\typrp}{(\subst{0}{2}{\tmabs{\tyind}{\tmapp{0}{2}} \listsep 1})} =
    \tmabs{\typrp}{\tmabs{\tyind}{\tmapp{0}{2}}} \\
    %
    \subst{(\tmabs{\typrp}{0})}{0}{\tmabs{\tyind}{\tmapp{0}{0}}} =
    \tmabs{\typrp}{(\subst{0}{1}{\tmabs{\tyind}{\tmapp{1}{1}} \listsep 0})} =
    \tmabs{\typrp}{\tmabs{\tyind}{\tmapp{1}{1}}} \\
\end{gather*}

In a substitution $\subst{t}{m}{s_0, \dots, s_n}$, the resulting term and all the terms $s_0, \dots, s_n$ are meant to be interpreted with $m$ free variables, while the term $t$ is meant to be interpreted with $n+1$ free variables.

\begin{theorem}
    Let $\dotlist{T}{1}{m}$ and $\dotlist{S}{1}{n}$ be types and let $t$ and $\dotlist{s}{1}{n}$ be terms such that $\hastype{\listdelim{T_1 \listsep \dots \listsep T_m}}{s_i}{S_i}$ for every $i = 1, \dots, n$.
    If $\hastype{\listdelim{S_1 \listsep \dots \listsep S_n}}{t}{T}$ then $\hastype{\listdelim{T_1 \listsep \dots \listsep T_m}}{\subst{t}{m}{s_1 \listsep \dots \listsep s_n}}{T}$.
\end{theorem}

\begin{theorem}
    Let $\dotlist{T}{1}{m}$ and $\dotlist{S}{1}{n}$ be types and let $t$ and $\dotlist{s}{1}{n}$ be terms such that $\hastype{\listdelim{T_1 \listsep \dots \listsep T_m}}{s_i}{S_i}$ for every $i = 1, \dots, n$.
    If $m \le n$ and $\hastype{\listdelim{T_1 \listsep \dots \listsep T_m}}{\subst{t}{m}{s_1 \listsep \dots \listsep s_n}}{T}$
    then $\hastype{\listdelim{S_1 \listsep \dots \listsep S_n}}{t}{T}$.
\end{theorem}

For example, $\hastype{\listdelim{\typrp}}{0}{\typrp}$ and $\hastype{\listdelim{\tyind}}{\top}{\typrp}$ and $\hastype{\listdelim{\tyind}}{\subst{0}{1}{\top}}{\typrp}$, that is, $\hastype{\listdelim{\tyind}}{\top}{\typrp}$.

\begin{definition}
    We define the substitution of the term $s$ with contexts $m$ and $n$ into the term $t$ as:
    \[
        \inddef{\substl{m}{n}{s}{t}}{\subst{t}{n}{0 \listsep \dots \listsep m-1 \listsep s}}
    \]
\end{definition}


\subsection{Reduction}

\begin{definition}
    Given a binary relation on terms $R_m$, we define another relation $\step{R}{m}$ inductively as follows:
    \begin{gather*}
        \inferrule{R_m(a, b)}{\step{R}{m}{(a, b)}} \\
        %
        \inferrule{\step{R}{m}{(a, b)}}{\step{R}{m}{(\tmapp{a}{t}, \tmapp{b}{t})}} \qquad
        %
        \inferrule{\step{R}{m}{(a, b)}}{\step{R}{m}{(\tmapp{s}{a}, \tmapp{s}{b})}} \\
        %
        \inferrule{\step{R}{m+1}{(a, b)}}{\step{R}{m}{(\tmabs{T}{a}, \tmabs{T}{b})}}
    \end{gather*}
\end{definition}

\begin{definition}
    We define the binary relation $\betaconv{m}{}$ between terms inductively as follows:
    \[
        \inferrule{ }{\betaconv{m}{\bigl(\tmapp{(\tmabs{T}{t})}{s}, \substl{m}{m}{s}{t}\bigr)}}
    \]
\end{definition}

\begin{definition}
    We define the binary relation $\etaconv{m}{}$ between terms inductively as follows:
    \[
        \inferrule{ }{\etaconv{m}{(\tmabs{T}{\tmapp{t}{m}}, \decrbound{t})}} \infcondr{if $t$ does not contain $m$}
    \]
\end{definition}

We define $\betastep{m}{}$ as $\step{\betaconv{m}{}}{}$ and $\etastep{m}{}$ as $\step{\etaconv{m}{}}{}$.

\subsection{Nominal constants}

For every type $T$ let $\pi_T \colon \mathbb{N} \to \mathbb{N}$.
The application of $\pi$ to a term is defined recursively as follows
\begin{gather*}
    \inddef{\nommap{\pi}{\tmnom{T}{n}}}{\tmnom{T}{(\pi_T (n))}} \\
    \inddef{\nommap{\pi}{(\tmabs{T}{t})}}{\tmabs{T}{(\nommap{\pi}{t})}} \\
    \inddef{\nommap{\pi}{(\tmapp{s}{t})}}{\tmapp{(\nommap{\pi}{s})}{(\nommap{\pi}{t})}} \\
    \inddef{\nommap{\pi}{t}}{t} \infcondr{in all other cases}
\end{gather*}

We say that two terms $a, b$ are equivalent, written $a \approx b$, if there is a family of bijective functions $\pi_T$ such that $a$ is $\lambda$-equivalent to $\nommap{\pi}{b}$.

We define the support of a term $t$, indicated as $\supp{t}$, as the set of nominal constants appearing in $t$.


\section{G Logic}
